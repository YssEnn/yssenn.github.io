ORC文字识别

2023/05/12 19:00 下午

docker

下载镜像

docker pull registry.baidubce.com/paddlepaddle/paddle:2.4.1

运行镜像

docker run -p 2003:8080 -v /mnt/paddle:/mnt/paddle --privileged=true --name paddle -d -t registry.baidubce.com/paddlepaddle/paddle:2.4.1

创建个文件orc.py

from paddleocr import PaddleOCR
import json
import uuid
import os
import flask
from flask import request
from flask_cors import *
server = flask.Flask(__name__)
CORS(server, supports_credentials=True)

path_base = '/mnt/paddle/upload/'


@server.route("/orc/upload", methods=["POST"])
def down():
    """接受前端传送过来的文件"""
    path = path_base+str(uuid.uuid1())+'/'
    path = path.strip()
    path = path.rstrip("\\")
    os.makedirs(path)
    file_obj = request.files.get("file")
    if file_obj is None:
        return "文件上传为空"

    # 直接使用文件上传对象保存
    file_obj.save(path+file_obj.filename)
    res = orc_run(path+file_obj.filename)
    da = {"code": "200", "msg": "ok", "data": res}
    return json.dumps(da, ensure_ascii=False)


def orc_run(path):
    # Paddleocr目前支持的多语言语种可以通过修改lang参数进行切换
    # 例如`ch`, `en`, `fr`, `german`, `korean`, `japan`
    ocr = PaddleOCR(use_angle_cls=True)  # need to run only once to download and load model into memory
    img_path = r''+path
    result = ocr.ocr(img_path, cls=True)
    data = []
    for idx in range(len(result)):
        res = result[idx]
        for line in res:
            print(line[1])
            data.append(line[1])
    return data


if __name__ == '__main__':
    server.run(debug=False, port=8080, host='0.0.0.0')
