# @author  : Yangshuaishuai

from sklearn.linear_model import LinearRegression
import sys
def Linear(XStr,YStr,XPredict):
	x = eval(XStr)
	y = eval(YStr)
	model = LinearRegression()
	model.fit(x, y)
	return model.predict([[int(XPredict)]])[0][0]
if __name__ == '__main__':
	try:
		XStr = sys.argv[1]
		YStr = sys.argv[2]
		X = sys.argv[3]
		Y = Linear(XStr, YStr, X)
		print("{\"code\":0,\"val\":"+str(Y)+"}")
	except Exception as e:
		print("{\"code\":1,\"error\":\""+str(e)+"\"}")

二次多项式

import numpy as np
from sklearn.linear_model import LinearRegression
import sys
def SecondTimePolynomial(XStr,YStr,XPredict):
    X = eval(XStr)
    Y = eval(YStr)
    x = []
    y = []
    # X，Y转为float浮点类型 然后防进 x,y中 因为下方是需要两种不同格式的 x，y
    for i in range(len(X)):
        X[i][0] = float(X[i][0])
        Y[i][0] = float(Y[i][0])
        x.append(X[i][0])
        y.append(Y[i][0])
    X = np.array(eval(str(X)))  # 将X转为2D数组 [[2000],[2001]] 转 [[2000][2001]]
    x2 = np.hstack([X, X ** 2])  # 这里给样本X再引入1个特征项，现在的特征就有2个
    model = LinearRegression()
    model.fit(x2, y)  # 将x平方带进去
    return model.predict([[int(XPredict), int(XPredict) ** 2]])[0]
if __name__ == '__main__':
	try:
		XStr = sys.argv[1]
		YStr = sys.argv[2]
		X = sys.argv[3]
		Y = SecondTimePolynomial(XStr, YStr, X)
		print("{\"code\":0,\"val\":"+str(Y)+"}")
	except Exception as e:
		print("{\"code\":1,\"error\":\""+str(e)+"\"}")
